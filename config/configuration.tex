\makeatletter
\newenvironment{explain_abstract}{% define abstract, not using the name abstract as usually already defined.
    \cleardoublepage
    \null\vfil
    \@beginparpenalty\@lowpenalty
    \begin{center}%
        \bfseries \abstractname
        \@endparpenalty\@M
    \end{center}}%
    {\par\vfil\null}
\makeatother


\newenvironment{acknowledgements}{
    \renewcommand\abstractname{Acknowledgements}
    \begin{explain_abstract}} {\end{explain_abstract}
}
